import glob
import librosa
import pyworld
import numpy as np
import matplotlib.pyplot as plt
from scipy.stats import norm
import IPython.display
from sklearn.mixture import GaussianMixture

def getWaves(gpath: str):
    return [librosa.load(path, sr=16000)[0].astype(np.double) for path in glob.glob(gpath)]

def getF0(waves):
    return [pyworld.harvest(w, 16000)[0] for w in waves]

def f0Conversion(f0, gmm1, gmm2, N, liner=10):
    gd = []
    for i in range(N):
        print((gmm.weights_[i] * norm.pdf(f0, loc=gmm1.means_[i, -1], scale=np.sqrt(gmm1.covariances_[i]))).max())
        s1 = norm.cdf(f0, loc=gmm1.means_[i, -1], scale=np.sqrt(gmm1.covariances_[i]))
        s2 = norm.ppf(s1, loc=gmm2.means_[i, -1], scale=np.sqrt(gmm2.covariances_[i]))
#         # 部分的に線形変換
        s2[s2==np.inf] = s2[s2!=np.inf].max()
        s2[s2==-np.inf] = s2[s2!=np.inf].min()
        gd.append(s2)
    return np.sum(gd, axis=0) / N
